def safeDiv (a b : Nat) : Option Nat :=
  if b = 0 then
    none
  else
    some (a / b)

theorem safeDiv_correct (a b : Nat) :
  (b = 0 ↔ safeDiv a b = none) ∧
  (b ≠ 0 ↔ safeDiv a b = some (a / b)) := by
  cases b
  · simp [safeDiv]
  · simp [safeDiv]
